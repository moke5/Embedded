24. 利用***\*位运算\****，编写一个函数，将一个整数的所有位逆序。 (8分)

\``` 

unsigned int reverseBits(unsigned int num) 

{ 

 

  unsigned int result = 0;

  int bitCount = sizeof(num) * 8; // 计算整数的位数

 

  // 逆序每一位

  for (int i = 0; i < bitCount; i++) {

​    // 将num的最后一位移动到result的最前面

​    result <<= 1; // result左移一位，为新的位腾出空间

​    result |= (num & 1); // 将num的最后一位（num & 1）放到result的最前面

​    num >>= 1; // num右移一位，移除已经处理的位

  }

  return result;

 

}

 \```
